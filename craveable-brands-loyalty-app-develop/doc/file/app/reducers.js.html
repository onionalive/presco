    import { combineReducers } from 'redux';

/* reducer-import */
import { DeleteViewReducer } from './components/deleteview/DeleteViewReducer';
import { OfferDetailReducer } from './components/offerdetail/OfferDetailReducer';
import { LandscapeReducer } from './components/landscape/LandscapeReducer';
import { SwiperContainerReducer } from './components/swipercontainer/SwiperContainerReducer';
import { MoreReducer } from './components/more/MoreReducer';
import { OffersReducer } from './components/offers/OffersReducer';
import { ProfileReducer } from './components/profile/ProfileReducer';
import { HomeReducer } from './components/home/HomeReducer';
import { AltReducer } from './components/alt/AltReducer';

/* navigation-reducers */
import NavigatorTabOne from './router/tabone/TabOneNavigation';
import NavigatorTabTwo from './router/tabtwo/TabTwoNavigation';
import NavigatorTabThree from './router/tabthree/TabThreeNavigation';
import NavigatorTabFour from './router/tabfour/TabFourNavigation';
import TabBar from './router/tabbar/TabBarNavigation';
import TransNav from './router/transitionrtl/TransitionNavigation';
import AppNav from './router/appnav/AppNavigation';

export default combineReducers({
	/* reducer-name */
	DeleteViewReducer,
	OfferDetailReducer,
	LandscapeReducer,
	SwiperContainerReducer,
	MoreReducer,
	OffersReducer,
	ProfileReducer,
	HomeReducer,
	AltReducer,
	appNav: (state,action) => AppNav.router.getStateForAction(action,state),
	transNav: (state,action) => TransNav.router.getStateForAction(action,state),
	tabBar: (state,action) => TabBar.router.getStateForAction(action,state),
	tabBar: (state,action) => TabBar.router.getStateForAction(action,state),
	tabOne: (state,action) => NavigatorTabOne.router.getStateForAction(action,state),
	tabTwo: (state,action) => NavigatorTabTwo.router.getStateForAction(action,state),
	tabThree: (state,action) => NavigatorTabThree.router.getStateForAction(action,state),
	tabFour: (state,action) => NavigatorTabFour.router.getStateForAction(action,state)
});
